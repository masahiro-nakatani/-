# SourceForge.JPからファイル取得するための一時的な関数を定義

# ソースファイルを取得し，展開
  # cd /tmp && pwd
  /tmp
  # sfget http://sourceforge.jp/projects/gnupack/downloads/46325/emacs-23.4.tar.gz/
  # sfget http://sourceforge.jp/projects/gnupack/downloads/54283/emacs-23.4-ime-2012-02-05.patch.tar.gz/
  # sfget http://sourceforge.jp/projects/gnupack/downloads/46325/emacs-icon-2011-08-15.tar.gz/
  # tar zxf emacs-23.4.tar.gz
  # tar zxf emacs-23.4-ime-2012-02-05.patch.tar.gz
  # tar zxf emacs-icon-2011-08-15.tar.gz

# パッチ適用
  # cd emacs-23.4 && pwd
  /tmp/emacs-23.4
  # cp ../emacs.ico nt/icons
  # patch -b -p1 < ../emacs-23.4-ime-2012-02-05.patch
  # cd nt && pwd
  /tmp/emacs-23.4/nt

# ビルド実行
  # prefix=/app/emacs/emacs-23.4
  # ./configure.bat \
        --no-debug --with-gcc --enable-w32-ime \
        --cflags -O2 \
        --cflags -fno-omit-frame-pointer \
        --cflags -funsigned-char \
        --cflags -mthreads \
        --cflags -pipe \
        --cflags -DRECONVERSION \
        --cflags -DIME_UNICODE \
        --cflags -I${INST_DIR/\\/\/}/app/mingw/include \
        --ldflags -L${INST_DIR/\\/\/}/app/mingw/lib
  # mingw32-make bootstrap

# インストール
  # mingw32-make install
  # cp ${INST_DIR/\\/\/}/app/mingw/dll/* "$prefix/bin"
